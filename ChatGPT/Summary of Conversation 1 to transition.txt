6) src/app/api/ingest/route.ts (secure + upserts)

(abbrev; key pieces)

import { NextResponse } from 'next/server';
import { getPrisma } from '@/lib/db';

export async function POST(req: Request) {
  // Auth via header
  const key = process.env.INGEST_KEY;
  const provided = req.headers.get('x-ingest-key');
  if (key && provided !== key) {
    return NextResponse.json({ ok:false, error:'Unauthorized' }, { status: 401 });
  }

  const prisma = getPrisma();
  const { provider, listing, variants = [] } = await req.json();

  // upsert provider
  const prov = await prisma.provider.upsert({
    where: { website: provider.website },
    update: { name: provider.name, phone: provider.phone ?? undefined, email: provider.email ?? undefined },
    create: { name: provider.name || new URL(provider.website).hostname, website: provider.website, phone: provider.phone ?? undefined, email: provider.email ?? undefined }
  });

  // upsert listing by sourceUrl
  const lst = await prisma.listing.upsert({
    where: { sourceUrl: listing.sourceUrl },
    update: { providerId: prov.id, ...listing },
    create: { providerId: prov.id, ...listing }
  });

  // one default variant
  if (variants[0]) {
    const v0 = variants[0];
    await prisma.tripVariant.upsert({
      where: { id: `${lst.id}-default` },
      update: { durationHours: v0.durationHours, isPrivate: v0.isPrivate ?? true, priceFrom: v0.priceFrom, priceUnit: v0.priceUnit ?? 'trip' },
      create: { id: `${lst.id}-default`, listingId: lst.id, durationHours: v0.durationHours, isPrivate: v0.isPrivate ?? true, priceFrom: v0.priceFrom, priceUnit: v0.priceUnit ?? 'trip' }
    });
  }

  return NextResponse.json({ ok:true, providerId: prov.id, listingId: lst.id });
}

7) src/app/api/listings/route.ts (filters)

(abbrev; key pieces)

import { NextResponse } from 'next/server';
import { getPrisma } from '@/lib/db';

export const dynamic = 'force-dynamic';
export const revalidate = 0;
export const runtime = 'nodejs';

export async function GET(req: Request) {
  const prisma = getPrisma();
  const { searchParams } = new URL(req.url);
  const q = searchParams.get('q')?.trim() || undefined;
  const city = searchParams.get('city')?.trim() || undefined;
  const state = searchParams.get('state')?.trim() || undefined;
  const minPrice = Number(searchParams.get('minPrice') || '') || undefined;
  const maxPrice = Number(searchParams.get('maxPrice') || '') || undefined;
  const limit = Math.min(Number(searchParams.get('limit') || 24), 50);
  const offset = Math.max(Number(searchParams.get('offset') || 0), 0);

  const where:any = {};
  if (city) where.city = { equals: city, mode: 'insensitive' };
  if (state) where.state = { equals: state, mode: 'insensitive' };
  if (q) where.OR = [
    { title: { contains: q, mode: 'insensitive' } },
    { description: { contains: q, mode: 'insensitive' } },
    { city: { contains: q, mode: 'insensitive' } },
    { species: { hasSome: [q] } }
  ];

  const variantWhere:any = {};
  if (minPrice !== undefined) variantWhere.priceFrom = { ...(variantWhere.priceFrom||{}), gte: minPrice };
  if (maxPrice !== undefined) variantWhere.priceFrom = { ...(variantWhere.priceFrom||{}), lte: maxPrice };

  const listings = await prisma.listing.findMany({
    where,
    include: {
      provider: true,
      variants: {
        where: Object.keys(variantWhere).length ? variantWhere : undefined,
        orderBy: { priceFrom: 'asc' }, take: 1
      }
    },
    orderBy: [{ createdAt: 'desc' }], take: limit, skip: offset
  });

  return NextResponse.json(listings);
}

🧯 Common gotchas (fast fixes)

Prisma error “client not initialized” on Vercel → ensure postinstall: prisma generate and build: prisma generate && next build.

Prerender error for API → add export const dynamic = 'force-dynamic' + runtime = 'nodejs' at top of API routes; don’t use output: 'export'.

DIRECT_URL missing → Prisma reads .env by default; ensure both URLs are in .env (not only .env.local).

Browser stuck on old Vercel URL → clear Edge cached redirects for your domain (you did this once).
