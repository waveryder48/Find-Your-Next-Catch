## `src/app/page.tsx` (Homepage)

```tsx
import Link from "next/link";
import { prisma } from "@/lib/prisma";

const isUrl = (s?: string | null) =>
  !!s && (/^https?:\/\//i.test(s) || (/\.[a-z]{2,}($|[/?#])/i.test(s) && !/\s/.test(s)));

const toUrl = (s?: string | null) =>
  !isUrl(s) ? null : /^https?:\/\//i.test(s!) ? s! : `https://${s}`;

export default async function HomePage() {
  // Optional: show a few latest vessels. Safe fallback if DB/env is missing.
  let featured: Awaited<ReturnType<typeof prisma.listing.findMany>> = [];
  try {
    featured = await prisma.listing.findMany({
      take: 6,
      include: { provider: true },
      orderBy: { createdAt: "desc" },
    });
  } catch {
    featured = [];
  }

  return (
    <main className="mx-auto max-w-6xl p-6">
      {/* Hero */}
      <section className="flex flex-col items-center text-center py-16">
        <h1 className="text-6xl font-extrabold tracking-tight mb-4 drop-shadow">
          Find Your Next Catch
        </h1>
        <p className="text-lg mb-8">Browse sportfishing vessels across Southern California.</p>

        <Link
          href="/listing"
          className="px-6 py-3 border border-black bg-white text-black rounded-lg shadow hover:bg-gray-50 transition"
        >
          Vessels
        </Link>
      </section>

      {/* Latest vessels (optional) */}
      {featured.length > 0 && (
        <section className="mt-10">
          <h2 className="text-xl font-semibold mb-4">Latest vessels</h2>
          <div className="grid gap-4 md:grid-cols-2 lg:grid-cols-3">
            {featured.map((r) => {
              const vesselUrl = toUrl(r.sourceUrl);          // external
              const landingUrl = toUrl(r.provider?.website); // external
              return (
                <div key={r.id} className="rounded-2xl p-4 shadow border bg-white/90">
                  <h3 className="font-medium">{r.title || "Untitled Vessel"}</h3>
                  <p className="text-sm text-gray-600">
                    {[r.city, r.state].filter(Boolean).join(", ")}
                  </p>

                  <div className="mt-3 flex flex-wrap gap-3">
                    {vesselUrl && (
                      <a href={vesselUrl} className="rounded-lg border px-3 py-1.5 hover:bg-gray-50">
                        Vessel Information
                      </a>
                    )}
                    {landingUrl && (
                      <a href={landingUrl} className="rounded-lg border px-3 py-1.5 hover:bg-gray-50">
                        Landing Website
                      </a>
                    )}
                  </div>
                </div>
              );
            })}
          </div>
        </section>
      )}
    </main>
  );
}
```

## `src/app/listing/page.tsx` (Listing index)

```tsx
import BackButton from "@/components/BackButton";
import { prisma } from "@/lib/prisma";

export const dynamic = "force-dynamic"; // avoid prerender failures

const isUrl = (s?: string | null) =>
  !!s && (/^https?:\/\//i.test(s) || (/\.[a-z]{2,}($|[/?#])/i.test(s) && !/\s/.test(s)));

const toUrl = (s?: string | null) =>
  !isUrl(s) ? null : /^https?:\/\//i.test(s!) ? s! : `https://${s}`;

export default async function ListingPage() {
  let data: Awaited<ReturnType<typeof prisma.listing.findMany>> = [];
  try {
    data = await prisma.listing.findMany({
      include: { provider: true },
      orderBy: { createdAt: "desc" },
      take: 60,
    });
  } catch {
    data = [];
  }

  return (
    <main className="mx-auto max-w-6xl p-6">
      <div className="mb-6">
        <BackButton />
      </div>

      <h1 className="text-3xl font-bold mb-4">Vessels</h1>

      {data.length === 0 ? (
        <p className="text-gray-700">No vessels available yet.</p>
      ) : (
        <div className="grid gap-4 md:grid-cols-2 lg:grid-cols-3">
          {data.map((r) => {
            const vesselUrl = toUrl(r.sourceUrl);
            const landingUrl = toUrl(r.provider?.website);

            return (
              <div key={r.id} className="rounded-2xl p-4 shadow border bg-white/90">
                <h3 className="font-medium">{r.title || "Untitled Vessel"}</h3>
                <p className="text-sm text-gray-600">
                  {[r.city, r.state].filter(Boolean).join(", ")}
                </p>

                <div className="mt-3 flex flex-wrap gap-3">
                  {vesselUrl && (
                    <a href={vesselUrl} className="rounded-lg border px-3 py-1.5 hover:bg-gray-50">
                      Vessel Information
                    </a>
                  )}
                  {landingUrl && (
                    <a href={landingUrl} className="rounded-lg border px-3 py-1.5 hover:bg-gray-50">
                      Landing Website
                    </a>
                  )}
                </div>
              </div>
            );
          })}
        </div>
      )}
    </main>
  );
}
```
